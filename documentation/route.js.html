/**
 * Holds route information from Router. For more information on routing, see Router.

 * @constructor

	this.path = null;
	this.verb = null;
	this.controllerConstructor = null;
	this.action = null;
	this.methodName = null;
	this.method = null;
	this.argumentNames = null;
	this.isView = false;
	this.template = null;
	this.templatePath = null;
}
